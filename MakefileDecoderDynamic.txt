all: str_modul.a decoder.o
	gcc decoder.o -o decoder \
	-L./ -lstr_modul.so -Wl, -rpath,./
str_modul.a: util.o EncryptFunction.o
	gcc -shared -o str_modul.so util.o EncryptFunction.o
	rm util.o
	rm EncryptFunction.o
util.o:
	gcc -c -fPIC util.c -o util.o
decoder.o:
	gcc -c decoder.c -o decoder.o -std=c99
EncryptFunction.o:
	gcc -c -fPIC EncryptFunction.c -o EncryptFunction.o -std=c99
